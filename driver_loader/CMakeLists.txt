cmake_minimum_required(VERSION 3.24)

# ---------------------------------------------------------------------------
# Generate the 256 numbered stubs at configure time.
# ---------------------------------------------------------------------------

set(GENERATED_STUBS "${CMAKE_CURRENT_BINARY_DIR}/nt_stubs_generated.cpp")

set(_lines "// AUTO-GENERATED by CMakeLists.txt – DO NOT EDIT.\n")
string(APPEND _lines "#include \"nt_stubs_internal.hpp\"\n")
string(APPEND _lines "#include <cstdio>\n")
string(APPEND _lines "#include <utility>  // std::unreachable\n\n")

set(_table "static const nt_stubs_internal::stub_fn_t s_stub_table[] = {\n")

foreach(i RANGE 255)
    string(APPEND _lines
        "[[gnu::noinline, gnu::used]] static void* nt_stub_${i}() noexcept {\n"
        "    volatile int idx = ${i};\n"
        "    std::fprintf(stderr, \"[nt_stubs] stub-enter #%d\\n\", idx);\n"
        "    std::fflush(stderr);\n"
        "    nt_stubs_internal::handle_call(idx);\n"
        "    std::fprintf(stderr, \"[nt_stubs] stub-exit  #%d -> <unreachable>\\n\", idx);\n"
        "    std::fflush(stderr);\n"
        "    std::unreachable(); }\n"
    )
    string(APPEND _table "    nt_stub_${i},\n")
endforeach()

string(APPEND _table "};\n\n")
string(APPEND _lines "${_table}")
string(APPEND _lines
    "namespace nt_stubs_internal {\n"
    "const stub_fn_t* get_stub_table() noexcept { return s_stub_table; }\n"
    "} // namespace nt_stubs_internal\n"
)

file(WRITE "${GENERATED_STUBS}" "${_lines}")

# ---------------------------------------------------------------------------
# driver_loader static library
# ---------------------------------------------------------------------------

add_library(driver_loader STATIC
    src/driver_loader.cpp
    src/nt_stubs.cpp
    "${GENERATED_STUBS}"
)

target_include_directories(driver_loader
    PUBLIC  include
    PRIVATE src
)

# The library needs to link against Windows system libraries for VirtualAlloc,
# HeapAlloc, WideCharToMultiByte, etc.
# These are already available via the MinGW sysroot import libraries.
target_link_libraries(driver_loader PUBLIC kernel32)
