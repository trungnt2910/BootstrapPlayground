cmake_minimum_required(VERSION 3.24)

# ---------------------------------------------------------------------------
# Generate the 256 numbered stubs at configure time.
# ---------------------------------------------------------------------------

set(STUB_SLOT_COUNT 256)

set(GENERATED_STUBS "${CMAKE_CURRENT_BINARY_DIR}/nt_stubs_generated.cpp")

set(_lines "// AUTO-GENERATED by CMakeLists.txt - DO NOT EDIT.\n")
string(APPEND _lines "#include \"nt_stubs_internal.hpp\"\n")
string(APPEND _lines "#include <print>\n")
string(APPEND _lines "#include <utility>  // std::unreachable\n\n")

set(_table "static constexpr STUB_FUNCTION NtStubTable[] =\n")
string(APPEND _table "{\n")

math(EXPR STUB_SLOT_MAX_INDEX "${STUB_SLOT_COUNT} - 1")
foreach(i RANGE ${STUB_SLOT_MAX_INDEX})
    string(APPEND _lines
        "static void* NtStub${i}() noexcept"
        "{\n"
        "    std::println(stderr, \"[nt_stubs] stub-enter ${i}\");\n"
        "    NtStubCall(${i});\n"
        "    std::println(stderr, \"[nt_stubs] stub-exit  ${i} -> <unreachable>\");\n"
        "    std::unreachable();\n"
        "}\n"
    )
    string(APPEND _table "    NtStub${i},\n")
endforeach()

string(APPEND _table "};\n\n")
string(APPEND _lines "${_table}")
string(APPEND _lines
    "const STUB_FUNCTION* NtStubGetTable() noexcept { return NtStubTable; }\n"
)

file(WRITE "${GENERATED_STUBS}" "${_lines}")

# ---------------------------------------------------------------------------
# driver_loader static library
# ---------------------------------------------------------------------------

add_library(driver_loader STATIC
    src/driver_loader.cpp
    src/nt_stubs.cpp
    "${GENERATED_STUBS}"
)

target_include_directories(driver_loader
    PUBLIC  "${CMAKE_CURRENT_SOURCE_DIR}/include"
    PRIVATE src
)

# The library needs to link against Windows system libraries for VirtualAlloc,
# HeapAlloc, WideCharToMultiByte, etc.
# These are already available via the MinGW sysroot import libraries.
target_link_libraries(driver_loader PUBLIC kernel32 dbghelp bcrypt)

target_compile_options(driver_loader PRIVATE -gcodeview -O0)
